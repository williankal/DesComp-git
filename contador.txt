#Laço principal

#checa limite se flag estiver ativa, ignora KEY0

LDA @6
CEQ @11
JEQ @100 #PULA KEY0

#rotina observa key0
LDA @352
CEQ @10 #checa se KEY0 foi pressionada
JEQ @100 #se não foi le key1(definir)
JSR @100 #se foi pula para incremento

#rotina KEY1
LDA @353
CEQ @10 #checa se KEY1 foi pressionada
JEQ @100 #se não foi le fpga reset(definir)
jsr @100 #pula para rotina de limite
JSR @100 #pula para LIMITE DE CONTAGEM

#Rotine fpga_reset
