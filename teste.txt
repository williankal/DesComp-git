#VARIAVEIS DE COMPARAÇÃO

#ARMAZENANDO LIMITES DE CONTAGEM

STA @31 #Armazena o limie de contagem em MEM[31] (dezenas)

STA @35 #Armazena o limie de contagem em MEM[35] (cent milhares)

#LOOP PRINCIPAL => CHECA KEY 0/1/FPGA
#CHECA FLAG
LDA @6 #Carrega acumulador com valor da flag MEM[6]
CEQ @11 #Compara se a flag MEM[6] está ativa
JEQ @41 #Se for igual, pula para o endereço (CHECA KEY1[pula key0])

#CHECA KEY 0
LDA @352 #Carrega acumulador com o valor de KEY0
CEQ @10 #Compara se o valor do acumulador é igual a 1 (Se key0 foi pressionado)
JEQ @42  #Se não for igual pula para key1
JSR @54 #Pula para a subrotina de incremento

#CHECA KEY 1

LDA @353 #Carrega acumulador com o valor de KEY1
CEQ @10 #Compara se o valor do acumulador é igual a 1 (Se key1 foi pressionado)
JEQ @45 #PULA PARA CHECA LIMITPULA PARA CHECA LIMITEE DE CONTAGEM(DEFINIR AINDA) 
JSR @156   #PULA PARA DEFINIR LIMITE
JSR @128 #pula para checar limite(definir ainda)

#CHECA FPGA_RESET

LDA @356 #Carrega acumulador com o valor de FPGA_RESET
CEQ @11 #Compara se o valor do acumulador é igual a 0 
JEQ @52 #Se foi pressionado pulta para atualiza display
JSR @116 #Pula para subrotina: RESETAR PLACA(DEFINIR AINDA)

#ATUALIZA DISPLAY
JSR @103 #Pula para subrotina: ATUALIZA DISPLAY(DEFINIR AINDA)
JMP @34 #Volta para o loop principal

#LOOP INCREMENTO 

STA @511 #Limpa KEY 0

#INCREMENTO UNIDADE
LDA @10 #Carrega acumulador com o valor de MEM[10](unidades)
SOMA @1 #Soma 1 ao valor das unidades
STA @0
CEQ @3
JEQ @59#Se for igual 10 pula para o endereço incrementar dezena(definir ainda)
RET @0 #VOLTA PRO LOOP PRINCIPAL


#INCREMENTO DEZENA

STA @0 #Salva o valor 0 em MEM[10]

LDA @11 #Carrega acumulador com o valor de MEM[11](dezenas)
SOMA @1 #Soma 1 ao valor da dezenas
STA @11
CEQ @3 #Compara o valor com 10
JEQ @67 #Se for igual a 10 pula para incremento centena(DEFINIR AINDA)
RET @0 #return

#INCREMENTO CENTENA

#INCREMENTO CENTENA MILHAR
